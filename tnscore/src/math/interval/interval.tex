\usepackage{mathtools}


\DeclarePairedDelimiterX{\interlike}[2]{\lbrack}{\rangle}{%
  #1 \,\delimsize|\, #2%
}

% \DeclareThreeDelimitersXXX
% \DeclareThreeDelimitersXnX

\NewDocumentCommand\DeclaireGenericInterval{mmmm}{
    \expandafter\DeclarePairedDelimiterX\csname #1\endcsname[2]{#2}{#4}{%
    ##1 \,\delimsize#3\, ##2%
  }
}

\NewDocumentCommand\DeclaireGenericIntervalBis{mmmm}{
    \expandafter\DeclarePairedDelimiterX\csname #1\endcsname[2]{#2}{#4}{%
    ##1 \,{#3}\, ##2%
  }
}

\interlike[\Big]{a}{b} \\
\interlike*{\frac{X^X}{p_p}}{z} \\
\interlike[\big]{z}{\frac{X^X}{p_p}}
\end{gather*}
