\documentclass[12pt, a4paper]{article}

\usepackage{mathtools}

\NewDocumentCommand\DeclareTripleDelimitersXXX{mmmm}{
    \expandafter\DeclarePairedDelimiterX\csname #1\endcsname[2]{#2}{#4}{%
    ##1 \,\delimsize#3\, ##2%
  }
}

\NewDocumentCommand\DeclareTripleDelimitersXnX{mmmm}{
    \expandafter\DeclarePairedDelimiterX\csname #1\endcsname[2]{#2}{#4}{%
    ##1 \,{#3}\, ##2%
  }
}

\begin{document}

\DeclareTripleDelimitersXXX{test}{\rbrack}{|}{\rangle}
$\test12$
$\test[\Big]12$
$\test*{\dfrac12}3$

\DeclareTripleDelimitersXnX{testb}{\rbrack}{|}{\rangle}
$\testb12$
$\testb[\Big]12$
$\testb*{\dfrac12}3$

\end{document}
